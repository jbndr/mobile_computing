\chapter{Grundlagen}

\begin{itemize}
	\item Grundbegriffe
	\item Einführung in grundlegenden Konzepte, Rahmenwerke, Architekturen, Methoden, Algorithmen, 
	\item Wichtige Standards, Technologien, Werkzeuge
	\item Beschreibung des betrachteten Szenarios
	\item ... 
\end{itemize}
